\section{Tensor Product of \( R \)-modules}
For simplicity, consider \( R \) as commutative ring. If \( M,N,P \) are \( R \)-modules, a map \(
\vp: M\times N \to P \) is \underline{bilinear} if \( \vp(u,-), \vp(-,v) \) are \( R \)-linear maps
\( \forall \; u \in M, v\in N \).

\begin{definition}
	A \underline{tensor product} \( M \otimes_R N \) is an \( R \)-module with a bilinear map \( \vp:
	M \times N \to M \otimes_R N\), written as \( \vp(u,v) = u\otimes v \), which is universal: \(
	\forall  \) bilinear map \( \psi : M\times N \to P \), \( \exists ! \) linear map
	\[
		f: M \otimes_R N \to P \quad s.t. \; f\circ \vp = \psi
	\]
\end{definition}

\begin{remark}
	As usual with universal properties, this characters \( M \otimes_R N \) up to a canonical
	isomorphpisms that is commutative with \( \vp \).
\end{remark}

\begin{remark}
	The proof of existence of \( M \otimes_R N \) follows as in the case \( R = \CC \): Let
	\[
		M \otimes_R N \coloneqq  \qo{R^{(I)}}{T}
	\]
	where \( R^{(I)} \) is the free \( R \)-module with basis \( \{e_{(u,v)} \; | \; (u,v) \in I\} \)
	and \( I = \{(u,v) \; | \; u\in M, v\in N\} \) and \( T \) is the submodule generated by all
	elements of the form
	\[
		e_{(a_1u_1+a_2u_2,v)} - a_1 e_{(u_1,v)} - a_2e_{(u_2,v)} \quad \forall \; u_1,u_2\in M, v\in N,
		a_1,a_2 \in R
	\]
	and similarly w.r.t. the second variable. And define
	\[
		\vp(u,v) \coloneqq  \overline{e_{(u,v)}}
	\]
\end{remark}

\begin{remark}
	It follows from this description, that every element of \( M \otimes_R N \) can be written as
	\[
		\sum_{i=1}^r a_i \overline{e_{(u_i,v_i)}} = \sum_{i=1}^r x_i \otimes y_i
	\]
	where \( x_i = a_i u_i, y_i = v_i \), and it is very far from unique.
\end{remark}

\begin{remark}[\textbf{General Properties}]
	\leavevmode
	\begin{enumerate}
		\item Functorial behabviors: If \( f: M \to M' \) and \( g: N \to N' \) are \( R \)-linear,
		      consider
		      \[
			      \begin{aligned}
				      M \times N \to M' \times N' & \to M' \otimes_R N'       \\
				      (u,v) \mapsto (f(u),g(v))   & \mapsto f(u) \otimes g(v)
			      \end{aligned}
		      \]
		      by the universal property of \( M \otimes_R N \), there exists a unique linear map
		      \[
			      \begin{aligned}
				      f\otimes g: M \otimes_R N            & \to M'\otimes_R N'  \\
				      \text{s.t. } f\otimes g (u\otimes v) & = f(u) \otimes g(v)
			      \end{aligned}
		      \]
		      this is functorial: \( f': M' \to M'' , g': N' \to N'' \) be \( R \)-linear, then
		      \[
			      (f'\otimes g') \circ (f\circ g) = (f'\circ f) \otimes (g'\circ g)
		      \]
		      it is enough to check that both siddes agree on elements of the form \( u\otimes v \).
		\item Commutativity of \( \otimes \): for every \( R \)-module \( M,N \), there exists a unique
		      isomorphism
		      \[
			      \begin{aligned}
				      M \otimes_R N & \xrightarrow{\sim} N \otimes_R M                    \\
				      u\otimes v    & \mapsto v \otimes u \quad \forall \; u\in M, v\in N
			      \end{aligned}
		      \]
		      \begin{proof}
			      We have
			      \[
				      \begin{cases}
					      M \times N \to N \times M \to N \otimes_R M \\
					      (u,v) \mapsto (v,u) \mapsto v\otimes u
				      \end{cases} \implies \text{by universal property}
			      \]
			      there exists a unique \( f: M\otimes_R N \to N \otimes_R M \) beiing \( R \)-linear, s.t.
			      \[
				      f(u\otimes v) = v\otimes u
			      \]
			      Similarly, there exists a unique \( R \)-linear map \( g: N\otimes_R M \to M \otimes_R N \),
			      s.t. \( g(v\otimes u) = u\otimes v \; \forall \; v\in M, u \in N\). It is easy to see
			      \[
				      \begin{aligned}
					      g\circ f & = \Id_{M\otimes_R N} \\
					      f\circ g & = \Id_{N\otimes_R M}
				      \end{aligned}
			      \]
		      \end{proof}

		\item Associativity of \( \otimes \): If \( M,N,P\) are \( R \)-modules, then there exists a
		      unique isomorphism of \( R \)-modules,
		      \[
			      \begin{aligned}
				      (M\otimes_R N) \otimes_R P & \xrightarrow{f} M \otimes_R (N\otimes_R P) \\
				      (u\otimes v) \otimes w     & \mapsto u \otimes (v\otimes w)
			      \end{aligned}
		      \]
		      \begin{proof}
			      Similar to the proof of 2.
		      \end{proof}

		\item For every \( R \)-module \( M \), we have an isomorphism
		      \begin{align*}
			      f: M & \xrightarrow{\sim} R \otimes_R M \\
			      f(u) & = 1\otimes u
		      \end{align*}

		      \begin{proof}
			      Consider the map
			      \begin{align*}
				      R\times M & \to M      \\
				      (a,x)     & \mapsto ax
			      \end{align*}
			      this map is bilinear, thus there exists a unqiue map
			      \[
				      \begin{aligned}
					      g: R \otimes_R M & \to M      \\
					      a\otimes u       & \mapsto au
				      \end{aligned}
			      \]
			      it is clear that \( g\circ f = \Id_M \), and see that
			      \begin{align*}
				      f\circ g(a\otimes u) & = f(au) = af(u) = a(1\otimes u) = a\otimes u \\
				      \implies f\circ g    & =\Id_{R \otimes_R M}
			      \end{align*}
		      \end{proof}
		\item If \( M \) is free with basis \( (e_i)_{i\in I} \) and \( N \) is free with basis \( (f_j)_{j\in J} \), then
		      \( M\otimes_R N \) is free with basis \( (e_i\otimes f_j)_{i\in I, j\in J} \). It is easy to
		      see that they span the whole module. For linear independence, can take the tensor product of
		      two projection map and verify the result.
	\end{enumerate}
\end{remark}

\begin{eg}[\textbf{Applications}]
	\textbf{Extension of Scalars}: Suppose \( f: R\to S \) be a ring homomorphism, with \( R \)
	commutative ring, we have a functor of restriction of scalars:
	\[
		F: S\text{-\underline{mod}} \to R\text{-\underline{mod}}
	\]
	If \( M \) is an \( S \)-module, then \( F(M) = M \) as an abelian group with scalar
	multiplication given by \( au = f(a)u \; \forall \; a\in R \). If \( f: M \to N \) is a morphism
	of \( S \)-modules, then \( g: F(M) \to F(N) \) is a morphism of \( R \)-modules, of course
	compatible.

	Now assume \( S \) is an \( R \)-algebra, there is a functor of extension of scalar in the other
	direction: if \( M \) is a \( R \)-module, then
	\begin{claim}
		there is an \( S \)-module structure on \( S\otimes_R M \), s.t.
		\[
			b(b'\otimes u) = (bb') \otimes_R u \quad \forall \; b,b'\in S, u\in M
		\]
	\end{claim}

			\]
		\end{proof}
	\end{proof}
